    <script>
        // --- Konfigurasi & State ---
        const EXPIRY_DAYS = 14;
        let currentEmail = '';
        let refreshInterval = null;
        let storedEmails = {}; // Objek untuk menyimpan email per alamat email

        // --- Inisialisasi ---
        document.addEventListener('DOMContentLoaded', () => {
            loadData();
            generateNewEmail();
            startAutoRefresh();
        });

        // --- Database Lokal (LocalStorage) ---
        function loadData() {
            const savedData = localStorage.getItem('tempMailData');
            if (savedData) {
                storedEmails = JSON.parse(savedData);
            }
        }

        function saveData() {
            localStorage.setItem('tempMailData', JSON.stringify(storedEmails));
        }

        function deleteAllData() {
            if(confirm("Apakah Anda yakin ingin menghapus semua riwayat email? Data tidak dapat dikembalikan.")) {
                localStorage.removeItem('tempMailData');
                storedEmails = {};
                showToast("Semua data berhasil dihapus.", "warning");
                renderEmailList([]); // Clear view
            }
        }

        // --- Fungsi Utama ---

        function generateNewEmail() {
            const domain = '1secmail.com';
            const chars = 'abcdefghijklmnopqrstuvwxyz0123456789';
            let username = '';
            
            for (let i = 0; i < 6; i++) {
                username += chars.charAt(Math.floor(Math.random() * chars.length));
            }
            
            currentEmail = `${username}@${domain}`;
            
            // Update UI
            document.getElementById('email-address').value = currentEmail;
            
            // Jika email ini sudah ada di database, tampilkan pesan info
            if (storedEmails[currentEmail]) {
                showToast(`Menggunakan email lama (${currentEmail})`, "warning");
            }

            // Reset Inbox View
            document.getElementById('email-list').innerHTML = `
                <div class="empty-state">
                    <div class="empty-icon">📭</div>
                    <p>Menunggu email masuk...</p>
                </div>
            `;
            document.getElementById('email-view-container').classList.remove('active');
            document.getElementById('empty-view').style.display = 'flex';
            
            // Trigger cek email langsung
            checkMail();
        }

        // 2. Fungsi Cek Inbox (API)
        async function checkMail() {
            if (!currentEmail) return;

            // Update status badge
            const statusText = document.getElementById('refresh-status');
            statusText.innerText = "Checking...";

            try {
                const response = await fetch(`https://www.1secmail.com/api/v1/?action=getMessages&login=${currentEmail.split('@')[0]}&domain=${currentEmail.split('@')[1]}`);
                
                if (!response.ok) throw new Error("Gagal mengambil data");

                const data = await response.json();
                
                // Simpan ke Database Lokal
                if (data && Array.isArray(data)) {
                    storedEmails[currentEmail] = data;
                    saveData();
                }

                renderEmailList(data);
                
            } catch (error) {
                console.error("Error checking mail:", error);
                // Jika error koneksi, coba render dari data lokal
                if (storedEmails[currentEmail]) {
                    renderEmailList(storedEmails[currentEmail]);
                    showToast("Menggunakan data offline (terputus)", "warning");
                } else {
                    showToast("Gagal memeriksa email.", "error");
                }
            } finally {
                statusText.innerText = "Live";
            }
        }

        // 3. Render Daftar Email
        function renderEmailList(emails) {
            const listContainer = document.getElementById('email-list');
            
            // Filter berdasarkan input pencarian
            const searchTerm = document.getElementById('search-input').value.toLowerCase();
            
            // Filter array dan simpan ke variabel sementara
            const filteredEmails = emails.filter(email => {
                const sender = (email.from || '').toLowerCase();
                const subject = (email.subject || '').toLowerCase();
                return sender.includes(searchTerm) || subject.includes(searchTerm);
            });

            // Jika kosong, set pesan default
            if (!filteredEmails || filteredEmails.length === 0) {
                if (listContainer.querySelector('.empty-state') === null) {
                     listContainer.innerHTML = `
                        <div class="empty-state" style="height: 100px;">
                            <p>0 email ditemukan.</p>
                        </div>
                    `;
                }
                return;
            }

            listContainer.innerHTML = '';

            filteredEmails.forEach(email => {
                const date = new Date(email.date);
                const dateStr = date.toLocaleString('id-ID', { 
                    day: '2-digit', month: 'short', hour: '2-digit', minute: '2-digit' 
                });

                const item = document.createElement('div');
                item.className = 'email-item';
                item.onclick = () => openEmail(email);

                item.innerHTML = `
                    <div class="email-sender">
                        <span>${escapeHtml(email.from)}</span>
                        <span class="email-time">${dateStr}</span>
                    </div>
                    <div class="email-subject">${escapeHtml(email.subject)}</div>
                    <div class="email-preview">${escapeHtml(email.text || email.html || 'Tidak ada isi teks')}</div>
                `;
                listContainer.appendChild(item);
            });
        }

        // 4. Buka Detail Email (Pemulihan dari LocalStorage)
        function openEmail(email) {
            // Set UI View
            document.getElementById('empty-view').style.display = 'none';
            const viewContainer = document.getElementById('email-view-container');
            viewContainer.classList.add('active');

            // Set Content
            document.getElementById('view-subject').textContent = email.subject;
            document.getElementById('view-sender').textContent = `Dari: ${email.from}`;
            
            const date = new Date(email.date);
            document.getElementById('view-date').textContent = date.toLocaleString('id-ID');

            // Set Body
            const bodyDiv = document.getElementById('view-body');
            
            if (email.html) {
                bodyDiv.innerHTML = email.html;
            } else if (email.text) {
                bodyDiv.textContent = email.text;
            } else {
                bodyDiv.textContent = "Tidak ada isi pesan.";
            }

            // Handle Attachment
            const attachArea = document.getElementById('view-attachment-area');
            const downloadLink = document.getElementById('download-link');

            if (email.attachments && email.attachments.length > 0) {
                attachArea.style.display = 'block';
                const att = email.attachments[0];
                const attUrl = `https://www.1secmail.com/api/v1/?action=download&login=${currentEmail.split('@')[0]}&domain=${currentEmail.split('@')[1]}&id=${email.id}&file=${att.id}`;
                
                downloadLink.href = attUrl;
                downloadLink.download = att.name;
                downloadLink.innerHTML = `⬇️ Unduh ${escapeHtml(att.name)}`;
            } else {
                attachArea.style.display = 'none';
            }

            // Highlight item active
            document.querySelectorAll('.email-item').forEach(el => el.classList.remove('active'));
            item.classList.add('active'); // Note: item is local scope here, ideally use ID
        }

        // --- Utilitas ---

        function copyEmail() {
            const copyText = document.getElementById("email-address");
            copyText.select();
            copyText.setSelectionRange(0, 99999);
            navigator.clipboard.writeText(copyText.value).then(() => {
                showToast("Email disalin!", "success");
            });
        }

        function refreshInbox() {
            checkMail();
        }

        function startAutoRefresh() {
            if (refreshInterval) clearInterval(refreshInterval);
            refreshInterval = setInterval(checkMail, 5000);
        }

        function filterEmails() {
            // Ambil email terbaru dari database terlebih dahulu
            let allEmails = storedEmails[currentEmail] || [];
            renderEmailList(allEmails);
        }

        function escapeHtml(text) {
            if (!text) return '';
            return text
                .replace(/&/g, "&amp;")
                .replace(/</g, "&lt;")
                .replace(/>/g, "&gt;")
                .replace(/"/g, "&quot;")
                .replace(/'/g, "&#039;");
        }

        function showToast(message, type = 'success') {
            const container = document.getElementById('toast-container');
            const toast = document.createElement('div');
            toast.className = `toast ${type}`;
            
            let icon = type === 'success' ? '✅' : (type === 'warning' ? '⚠️' : '❌');
            toast.innerHTML = `<span>${icon}</span> ${message}`;
            
            container.appendChild(toast);

            setTimeout(() => {
                toast.style.opacity = '0';
                setTimeout(() => toast.remove(), 300);
            }, 3000);
        }
    </script>
